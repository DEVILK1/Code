bfs加标记连通块，注意走过的连通块被标记后直接返回值，不用再走，所以vis数组不用每组数据都清0
```c++
#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

const int bx[] = {1, -1, 0, 0};
const int by[] = {0, 0, 1, -1};
const int MAZE = 1000 + 1;
int n, Q, start_x, start_y, ans, sign[MAZE][MAZE], rear, tail;
bool vis[MAZE][MAZE];
int que_x[MAZE * MAZE], que_y[MAZE * MAZE];
int a[MAZE][MAZE];

void bfs(int x, int y) {
	vis[x][y] = 1;
	rear = 1, tail = 1;
	que_x[tail] = x;
	que_y[tail] = y;
	while(rear <= tail) {
		for(int i = 0; i < 4; i++) {
			int xx = que_x[rear] + bx[i];
			int yy = que_y[rear] + by[i];
			if(xx >= 1 && xx <= n && yy >= 1 && yy <= n && !vis[xx][yy]) {
				if(a[xx][yy]^a[que_x[rear]][que_y[rear]]) {
					vis[xx][yy] = 1;
					que_x[++tail] = xx;
					que_y[tail] = yy;
				}
			}
		}
		rear++;
	}
}
int main() {
	scanf("%d%d", &n, &Q);
	for(int i = 1; i <= n; i++) {
		for(int j = 1; j <= n; j++) scanf("%1d", &a[i][j]);
	}
	while(Q--) {
		scanf("%d%d", &start_x, &start_y);
		if(sign[start_x][start_y]) {
			printf("%d\n", sign[start_x][start_y]);
			continue;
		}bfs(start_x, start_y);
		for(int i = 1; i <= tail; i++)
			sign[que_x[i]][que_y[i]] = tail;
		printf("%d\n", tail);
	}
}
```
